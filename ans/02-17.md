**static/script.js**

```js
function comment() {
  // コメントを取得する
  const comment = document.querySelector('#comment');

  // レビュー欄にdiv要素を追加する
  let review = document.querySelector("#review");
  review.innerHTML += `<div class="review-row">
                           <div>${comment.value}</div>
                           <hr>
                       </div>
                       `;
}
```
