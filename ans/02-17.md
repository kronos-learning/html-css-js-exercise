# JavaScript 演習問題

### 演習2.17（解答例）

**js-exercise13.html**

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>Exercise</title>
    <script type="text/javascript">
    function calc() {
        // それぞれのテキストボックスの入力値を取得する
        let num1 = document.getElementById('num1').value;
        let num2 = document.getElementById('num2').value;

        if (num1 === '' || num2 === '') {
            // いずれかの数値が未入力の場合、エラーメッセージを表示して処理を終了する
            alert('数値の入力が不正です。');
            return;
        }

        // 入力値を数値変換する
        num1 = parseInt(num1);
        num2 = parseInt(num2);

        // 結果div要素を取得する
        let resultDiv = document.getElementById('result');

        // h1要素を作成する
        let h1 = document.createElement('h1');

        // h1要素にテキストノード（加算結果）を子要素として追加する
        h1.appendChild(document.createTextNode(num1 + num2));

        // h1要素にstyle属性を追加する
        h1.setAttribute('style', 'color:red');

        // 結果div要素の最初の子要素とh1要素を置換する
        resultDiv.replaceChild(h1, resultDiv.firstChild);
    }
    </script>
</head>
<body>
    <div>
        <input type="number" id="num1" value="0">
        <span>+</span>
        <input type="number" id="num2" value="0">
        <button onclick="calc()">計算</button>
    </div>
    <div id="result">
    </div>
</body>
</html>
```
