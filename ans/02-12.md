# JavaScript 演習問題

### 演習2.12（解答例）

**js-exercise12.html**

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>Exercise</title>
    <script type="text/javascript">
    /*
     * 星を出力する処理
     */
    function output() {
        // input要素の入力値を取得する
        let count = document.getElementById('count').value;

        // 入力した数分ループする
        for (let i = 1; i <= count; i++) {
            // 星（空で初期化）
            let star = '';
            for (let j = 1; j <= i; j++) {
                // 行番号分、星を結合する
                star += '★';
            }
            console.log(star);
        }
    }
    </script>
</head>
<body>
    <div>
        <label for="count"></label>
        <input type="number" id="count" value="0">
        <button onclick="output()">出力</button>
    </div>
</body>
</html>
```
